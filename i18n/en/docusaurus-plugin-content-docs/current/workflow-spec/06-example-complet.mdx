# Complete example

Full pipeline for a COBOL module: dependency resolution, source upload, compilation, zUnit tests, SonarQube publishing, notification.

```yaml
apiVersion: automate4z/v1
kind: Workflow
name: build-and-test-cobol-module
description: |
  Full pipeline for a COBOL module: dependency resolution,
  source upload, compilation, zUnit unit tests, SonarQube
  publishing, completion notification.

a4z:
  version: '>=1.0 <2'

inputs:
  module:
    type: string
    required: true
    pattern: '^COBPRG[0-9]{2}$'
    description: Name of the COBOL module to compile.
  lpar:
    type: string
    default: dev01
    enum: [dev01, dev02]
    description: Target z/OS LPAR.
  publishSonar:
    type: boolean
    default: true

env:
  TZ: Europe/Paris

defaults:
  zos-jobs:
    lpar: ${{ inputs.lpar }}
    expectRC: 0
  zos-datasets:
    lpar: ${{ inputs.lpar }}

steps:
  - id: resolve-deps
    name: Resolve dependencies
    uses: z-deps
    with:
      manifest: ./z-manifest.yaml
      lockfile: ./z-deps.lock

  - id: prepare-build-dir
    name: Prepare build directory
    uses: fs
    with:
      action: mkdir
      path: ./build/${{ inputs.module }}
      recursive: true

  - id: upload-sources
    name: Upload sources
    needs: [resolve-deps]
    uses: zos-datasets
    with:
      action: upload
      localPath: ./src/cobol
      dataset: USER.${{ inputs.module }}.SRC

  - id: compile
    name: Compilation
    needs: [upload-sources]
    uses: zos-jobs
    with:
      jcl: ./jcl/compile.jcl
      vars:
        MODULE: ${{ inputs.module }}

  - id: check-rc
    name: Check return code
    needs: [compile]
    uses: assert
    with:
      condition: ${{ steps.compile.outputs.rc <= 4 }}
      message: 'Compilation failed with RC=${{ steps.compile.outputs.rc }}'

  - id: zunit
    name: zUnit unit tests
    needs: [check-rc]
    uses: zos-jobs
    with:
      jcl: ./jcl/zunit.jcl
      vars:
        MODULE: ${{ inputs.module }}

  - id: publish-sonar
    name: SonarQube publishing
    needs: [zunit]
    if: ${{ inputs.publishSonar == true }}
    run: |
      sonar-scanner \
        -Dsonar.projectKey=${{ inputs.module }} \
        -Dsonar.token=${{ secrets.SONAR_TOKEN }}

  - id: notify-done
    name: Completion notification
    needs: [zunit]
    if: ${{ always() }}
    uses: notify
    with:
      channel: email
      target: ['team-mainframe@corp.com']
      title: 'Build ${{ inputs.module }} - ${{ workflow.status }}'
      message: |
        Pipeline completed in ${{ workflow.duration }}.
        Compilation RC: ${{ steps.compile.outputs.rc }}
      severity: ${{ workflow.status == 'success' ? 'success' : 'error' }}
      fields:
        - { name: Module, value: '${{ inputs.module }}' }
        - { name: LPAR, value: '${{ inputs.lpar }}' }
        - { name: Build date, value: '${{ now("datetime") }}' }

outputs:
  jobId:
    description: Compilation job ID
    value: ${{ steps.compile.outputs.jobId }}
  rc:
    value: ${{ steps.compile.outputs.rc }}
```

## Execution

```bash
# Validate
a4z lint build-and-test-cobol-module.yml

# Show plan
a4z plan build-and-test-cobol-module.yml --input module=COBPRG01

# Run
a4z run build-and-test-cobol-module.yml \
  --input module=COBPRG01 \
  --input lpar=dev01 \
  --input publishSonar=true
```
